    function respond(message) {
      var responseDiv = document.getElementById("response");
      var heartDiv = document.getElementById("heart");
      
      if (message === 'TU me ama SIM') {
        responseDiv.innerHTML = "Tu me ama SIM";
        responseDiv.style.color = "#FF0000";
        heartDiv.style.display = "none";
      } else {
        responseDiv.innerHTML = message;
        responseDiv.style.color = "#000000";
        heartDiv.style.display = "block";
      }
    }
